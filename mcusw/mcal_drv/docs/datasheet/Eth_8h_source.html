1699 /*
1700  * \brief Dispatch virtual mac removal of given mac address from switch address resolution table request to ethernet firmware
1701  *
1702  * \verbatim
1703  * Service name : Eth_DispatchVirtmacDelAddr
1704  * Syntax : Std_ReturnType Eth_DispatchVirtmacDelAddr(uint8 CtrlIdx, uint8 *macAddress, uint32 vlanId)
1705  * Service ID[hex] : 0x27
1706  * Sync/Async : Asynchronous
1707  * Reentrancy : Non-Reentrant
1708  * Parameters (in) : CtrlIdx: Index of the controller within the context of the
1709  * Ethernet Driver
1710  * macAddress: Mac address to be deleted from the switch address resolution table
1711  * vlanId: Vlan Id associated with the mac Address. If vlan id is not needed, 0 should be passed
1712  * Parameters (inout): None
1713  * Parameters (out) : None
1714  * Return value : Std_ReturnType
1715  * E_OK: success
1716  * E_NOT_OK: transmission failed
1717  *
1718  * Description : The function dispatches RPC command to ethernet
1719  * firmware core to remove the given multicast macaddress
1720  * and vlan id from the switch address resolution table
1721  * On removal of entry switch would switch frames
1722  * received with the given destination macAddress:
1723  * vlan id combo by flooding to all the ports in vlan membership.
1724  * Address should have been added previously using Eth_DispatchVirtmacAddMcastAddr()/
1725  * Eth_DispatchVirtmacAddUnicastAddr()
1726  * If configured in virtual mac mode application should
1727  * invoke this API as follows:
1728  * 1. Eth_Init()
1729  * 2. Wait for fwRegisteredCb() callback
1730  * 3. Eth_DispatchVirtmacInit
1731  * 4. Wait for rpcCmdComplete() callback with status E_OK
1732  * 5. Eth_ControllerInit()
1733  * 6. Eth_DispatchVirtmacSubscribeDstMac()
1734  * 7. Wait for rpcCmdComplete() callback with status E_OK
1735  * 8. Eth_DispatchVirtmacAddMcastAddr()/Eth_DispatchVirtmacAddUnicastAddr()
1736  * 9. Wait for rpcCmdComplete() callback with status E_OK
1737  * 10. Eth_DispatchVirtmacDelAddr()
1738  * 11. Wait for rpcCmdComplete() callback with status E_OK
1739  * The callbacks fwRegisteredCb() and rpcCmdComplete() are
1740  * populated by the application as part of the Eth_ConfigType
1741  * structure
1742  * \endverbatim
1743  */
1744 FUNC(Std_ReturnType, ETH_CODE)
1745 Eth_DispatchVirtmacDelAddr(uint8 CtrlIdx, uint8 *macAddress, uint32 vlanId);
