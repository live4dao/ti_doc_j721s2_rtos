608 static inline void Vhwa_m2mVissParamsInit(Vhwa_M2mVissParams *prms)
609 {
610  if (NULL != prms)
611  {
612  prms->inputMode = VHWA_M2M_VISS_MODE_SINGLE_FRAME_INPUT;
613  prms->inFmt.width = 1280U;
614  prms->inFmt.height = 720U;
615  prms->inFmt.pitch[0] = 1280U * 2U;
616  prms->inFmt.dataFormat = FVID2_DF_RAW;
617  prms->inFmt.ccsFormat = FVID2_CCSF_BITS12_UNPACKED16;
618 
619  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].enable = (uint32_t)TRUE;
620  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.width = 1280U;
621  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.height = 720U;
622  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.pitch[0U] = 1280U;
623  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.pitch[1U] = 1280U;
624  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.dataFormat =
625  FVID2_DF_YUV420SP_UV;
626  prms->outPrms[VHWA_M2M_VISS_OUT_YUV420_8B_IDX].fmt.ccsFormat =
627  FVID2_CCSF_BITS8_PACKED;
628 
629  prms->enableGlbce = (uint32_t)FALSE;
630  prms->enableNsf4 = (uint32_t)FALSE;
631 #if defined VHWA_VPAC_IP_REV_VPAC3
632  prms->enableCac = (uint32_t)FALSE;
633 #endif
634  prms->edgeEnhancerMode = VHWA_M2M_VISS_EE_DISABLE;
635  prms->enableDpc = (uint32_t)FALSE;
636  prms->chromaMode = FCP_CHROMA_MODE_420;

640 static inline void Vhwa_m2mVissSl2ParamsInit(Vhwa_M2mVissSl2Params *prms)
641 {
642  uint32_t cnt;
643 
644  if (NULL != prms)
645  {
646  for (cnt = 0U; cnt < VHWA_M2M_VISS_MAX_INPUTS; cnt ++)
647  {
648  prms->maxInWidth[cnt] = 1920U;
649  }
650  prms->inCcsf = FVID2_CCSF_BITS12_UNPACKED16;
651  prms->inDepth = 3U;
652  for (cnt = 0U; cnt < VHWA_M2M_VISS_MAX_OUTPUTS; cnt ++)
653  {
654  if (VHWA_M2M_VISS_OUT_H3A_IDX == cnt)
655  {
656  prms->maxOutWidth[cnt] = 1024U;
657  }
658  else
659  {
660  prms->maxOutWidth[cnt] = 1920U;
661  }
662 
663 #if defined VHWA_VPAC_IP_REV_VPAC
664  if (cnt < VHWA_M2M_VISS_OUT_YUV420_8B_IDX)
665  {
666  prms->outCcsf[cnt] = FVID2_CCSF_BITS12_UNPACKED16;
667  }
668  else
669  {
670  prms->outCcsf[cnt] = FVID2_CCSF_BITS8_PACKED;
671  }
672 #elif defined VHWA_VPAC_IP_REV_VPAC3
673  prms->outCcsf[cnt] = FVID2_CCSF_BITS12_UNPACKED16;
674 #endif
675  prms->outDepth[cnt] = 2U;
676  }
677  }
678 }
