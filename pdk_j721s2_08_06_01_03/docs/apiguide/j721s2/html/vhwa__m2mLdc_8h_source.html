322 static inline void Vhwa_M2mLdcSl2AllocPrmsInit(Vhwa_M2mLdcSl2AllocPrms *sl2Prms)
323 {
324  uint32_t cnt;
325 
326  if (NULL != sl2Prms)
327  {
328  for (cnt = 0u; cnt < LDC_MAX_OUTPUT; cnt ++)
329  {
330  if (0u == cnt)
331  {
332  sl2Prms->enableOutput[cnt] = (uint32_t)TRUE;
333  sl2Prms->outCcsf[cnt] = FVID2_CCSF_BITS16_PACKED;
334  }
335  else
336  {
337  sl2Prms->enableOutput[cnt] = (uint32_t)TRUE;
338  sl2Prms->outCcsf[cnt] = FVID2_CCSF_BITS8_PACKED;
339  }
340  sl2Prms->maxNumBlocks[cnt] = 2U;
341  }
342  sl2Prms->maxBlockWidth = 64U;
343  sl2Prms->maxBlockHeight = 64U;
344  }
345 }
