398 static inline void Vhwa_M2mMscSl2AllocPrmsInit(Vhwa_M2mMscSl2AllocPrms *sl2Prms)
399 {
400  uint32_t cnt, idx;
401 
402  if (NULL != sl2Prms)
403  {
404  for(cnt = 0; cnt < VHWA_M2M_MSC_MAX_INST; cnt++)
405  {
406  for(idx = 0; idx < VHWA_M2M_MSC_MAX_IN_CHANNEL; idx++)
407  {
408  sl2Prms->maxInWidth[cnt][idx] = 1920U;
409  sl2Prms->inCcsf[cnt][idx] = FVID2_CCSF_BITS12_UNPACKED16;
410  sl2Prms->inBuffDepth[cnt][idx] = 6U;
411  }
412  }
413 
414  for(cnt = 0; cnt < MSC_MAX_OUTPUT; cnt++)
415  {
416  sl2Prms->maxOutWidth[cnt] = 1920U;
417  sl2Prms->outCcsf[cnt] = FVID2_CCSF_BITS12_UNPACKED16;
418  sl2Prms->outBuffDepth[cnt] = 2U;
419  }
420  }
421 }
