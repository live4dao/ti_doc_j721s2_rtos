468 static inline void Dss_rmInfoInit(Dss_RmInfo *rmInfo)
469 {
470  uint32_t i = 0U;
471  if(NULL != rmInfo)
472  {
473  for(i=CSL_DSS_COMM_REG_ID_0; i<CSL_DSS_COMM_REG_ID_MAX; i++)
474  {
475  rmInfo->isCommRegAvailable[i] = TRUE;
476  }
477  for(i=CSL_DSS_VID_PIPE_ID_VID1; i<CSL_DSS_VID_PIPE_ID_MAX; i++)
478  {
479  rmInfo->isPipeAvailable[i] = TRUE;
480  }
481  for(i=CSL_DSS_OVERLAY_ID_1; i<CSL_DSS_OVERLAY_ID_MAX; i++)
482  {
483  rmInfo->isOverlayAvailable[i] = TRUE;
484  }
485  for(i=CSL_DSS_VP_ID_1; i<CSL_DSS_VP_ID_MAX; i++)
486  {
487  rmInfo->isPortAvailable[i] = TRUE;
488  }
489  for(i=CSL_DSS_WB_PIPE_ID_1; i<CSL_DSS_WB_PIPE_ID_MAX; i++)
490  {
491  rmInfo->isWbPipeAvailable[i] = TRUE;
492  }
493  }
494 }
